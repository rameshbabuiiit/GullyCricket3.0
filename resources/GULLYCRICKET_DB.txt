CREATE TABLE gully_cricket (
	innings_id VARCHAR(1000) NOT NULL,
	score VARCHAR(1000) NULL DEFAULT NULL,
	wickets INT(4) NULL DEFAULT NULL,
	run_rate VARCHAR(10) NULL DEFAULT NULL,
	overs INT(4) NOT NULL DEFAULT '0',
	ball_by_ball VARCHAR(20000) NULL DEFAULT NULL,
	batsmen_Score VARCHAR(10000),
	last_updated_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	created_ts TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (innings_id)
);
